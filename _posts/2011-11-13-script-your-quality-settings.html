--- layout: post title: Script Your Quality Settings date: '2011-11-13T14:06:48-08:00' tags: [] tumblr_url: https://fugugames.tumblr.com/post/110515792081/script-your-quality-settings --- I have two problem with the Unity Quality Settings. It is inconvenient to change them per-platform and sometimes I want to tweak them per-scene instead of for the entire game. Fortunately, both issues can be resolved by scripting (that wasn’t always true - for a while, some of the newer additions to the Quality Settings were not accessable via script). For HyperBowl, I have a script that I attach to each scene with a property value for the shadow distance, since I want to vary that per scene. And I have settings that I want to vary per platform (or override, as in the case of fog on Android) Here, I also set my physics timestep - as with the quality settings, I worry about accidentally losing some of my preferred values, or even, in my more paranoid moments, that Unity has reset them. 
var shadowDistance:float = 80;

function Awake() {
    AdjustRenderSettings();
    AdjustPhysicsSettings();
}

function AdjustRenderSettings() {
// always vsync and use full texture resolution
    QualitySettings.masterTextureLimit = 0;
    QualitySettings.vSyncCount = 1;
#if UNITY_IPHONE || UNITY_ANDROID
    Application.targetFrameRate = 60;
    QualitySettings.pixelLightCount = 0;
    QualitySettings.anisotropicFiltering = AnisotropicFiltering.Enable;
    QualitySettings.antiAliasing = 0;
#endif
#if !UNITY_IPHONE && !UNITY_ANDROID
// apply per-scene shadow distance on desktop/web
    QualitySettings.shadowDistance = shadowDistance;
#endif
// override per-scene fog on Android
#if UNITY_ANDROID
    RenderSettings.fog = false;
#endif
}

function AdjustPhysicsSettings() {
    Time.fixedDeltaTime = 0.02;
}
