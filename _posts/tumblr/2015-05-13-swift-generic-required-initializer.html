More Swift madness. Without the required initializer, the generic built with that type generates bad_access crashes.


class WEObject {
   var ID:Int=0

   required init() {}

}

class Dict<T:WEObject> {

   var dict = [Int:T]()

   func Find(ID:Int)->T {
       if (dict[ID] == nil) {
           let route = T ()
           route.ID = ID
           dict[ID] = route
       }
       return dict[ID]!
   }
}
